# TAMAÑO BYTE
```asm
; Realice un programa que intercambie los contenidos de dos valores decimales definidos en dato1 y dato2                 

org 100h

; para operar como se trata de 8 bits, con un solo GPR es suficiente
mov al, [dato1]
mov ah, [dato2]                                                          

xchg al, ah  ; intercambia los las dos partes del mismo registro

mov [dato1], al
mov [dato2], ah             

hlt

dato1 db 254
dato2 db 255                                         
```

# TAMAÑO PALABRA
```asm
; Realice un programa que intercambie los contenidos de dos valores decimales definidos en dato1 y dato2                 

org 100h

; para operar como se trata de 16 bits, voy a necesitar dos GPR
mov ax, [dato1]
mov dx, [dato2]                                                          

xchg ax, dx  ; intercambia los valores entre dos registros

mov [dato1], ax
mov [dato2], dx
              
hlt
                                                                         
dato1 dw 255 ; esto entra un byte, pero 256 NO, asique para el intercambio debe ser de 16 bits
dato2 dw 256
```
