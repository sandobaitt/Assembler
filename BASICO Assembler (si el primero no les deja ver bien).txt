    == ASCCI ==

    ; valores mas usados son estos

        ; 10d es un salto de linea, se lo usa para bajar el cursor a la siguiente fila 
        ; 13d es un retorno de carro , se lo usa para mover el cursor al inicio (Enter)  
        ; 32d es un espacio, se lo usa para poner un caracter de espacio en blanco      
        ; 36d es un signo $, se lo usa para identifica un fin de cadena para el servicio 09h
        ; 48d-57d son los  Numeros '0' a '9'              
        ; 65d-90d son los Letras 'A' a 'Z'                
        ; 97d-122d son los Letras 'a' a 'z' 

        ; PRO TIPS:

        ; Para pasar de minuscula a MAYUSCULA: Resto 32 o 20h en hexa (sub al, 32 o sub al, 20h)
        ; Para pasar de MAYUSCULA a minuscula: Sumo 32 o 20h en hexa (add al, 32 o add al, 20h)
        ; De Caracter ASCII a Valor Numerico: Resto 48 o 30h en hexa (sub al, 48 o sub al, 30h)
        ; De Valor Numerico a Caracter ASCII: Sumo 48 o 30h en hexa (add al, 48 o add al, 30h)                 
        
    == Registros ==

    ; Registros generales (16 bits en 8086)
    ; Estos registros se pueden dividir en parte ALTA (High) y BAJA (Low)

    ; AX (ACUMULADOR) --> Operaciones aritmeticas                    
    ; BX (BASE)  --> Indice para direccionamiento de memoria  
    ; CX (CONTADOR) --> Contador para bucles (LOOP) y repeticiones
    ; DX (DATOS) --> Multiplicacion/Division y direcciones E/S
    ; Como seria esa division en parte alta y baja --> AX se divide en AH (bits 15-8) y AL (bits 7-0)

    ; Registros Indice
    ; Para trabajar con vectores y arreglos
    
    ; SI (FUENTE (Source)) --> Puntero de origen en copias de cadenas
    ; DI (DESTINO (Dest)) --> Puntero de destino en copias de cadenas  
   

    ; Registros Apuntadores (Control de Pila e Instruccion)
    
    ; BP (BASE (Base P.)) --> Apunta a la base de la pila (stack frame)
    ; SP  (PILA (Stack P.)) --> Apunta al tope de la pila (PUSH/POP)
    ; IP (INSTRUCCIÓN) --> Direccion de la siguiente instruccion
    

    ; Diferentes tamaños de registros (Visualizacion de bits)
    ; Para entender como los nombres cambian segun la capacidad:

        64 bits: [ RAX ] (Extendido - x64)
        32 bits: [  EAX  ] (Extendido - x86)
        16 bits: [   AX   ] <-- Este es el que usas en emu8086
        8 bits: [ AH | AL ] (Parte alta y baja)

    include ; Librerias

    .stack ; Segmento de pila (1024) por defecto

    .data ; Segmento de datos

        ; Todas las declaraciones -------------
        ;var 1
        ;var 2
        ;var 3
        ;var N

    .code ; Segmento de codigo

        ; Main
        main PROC

            ; Inicializacion del segmento de datos (obligatorio en .model small)
            mov ax, @data
            mov ds, ax

            Instruccion 1
            Instruccion 2

            Etiqueta1:
                Instruccion 3
                Instruccion 4
                Instruccion 5
                Instruccion 6

            Etiqueta2:
                Instruccion 7
                Instruccion 8

            Instruccion 9

        main ENDP   

    end main

    == 'emu8086.inc' ==

        ; Coleccion de macros (instrucciones que expanden a multiples lineas de codigo) y procedimientos que actuan 
        ;   como una capa de abstraccion sobre las interrupciones de BIOS y DOS. Su uso principal es reducir la complejidad 
        ;   visual del codigo y evitar errores comunes al manipular registros para tareas basicas de entrada y salida.

        ; Macros vs. Procedimientos
        ; Macros:

            ; Atajos que el ensamblador expande en el codigo. NUnca se usa la instrucción CALL con ellas

            print 'texto': Imprime una cadena en la posicion actual
            printn 'texto': Igual que PRINT, pero añade un salto de linea automatico
            putc char: Imprime un solo caracter (puede ser un registro como dl o una constante)
            gotoxy col, fila: Mueve el cursor a coordenadas especificas

        ; Procedimientos:

            ; Bloques de codigo mas complejos que viven al final del programa. SIEMPRE se usan con CALL para ejecutarlos y se definen al final del archivo

            scan_num: Lee un número del teclado y lo guarda en el registro cx
            print_num: Imprime el valor numerico contenido en el registro ax
            print_num_uns: Igual que el anterior, pero para numeros sin signo
            clear_screen: Borra toda la pantalla de la consola

        ; Para definir procedimientos o clear_screen, por ejemplo
        ; Se lo hace debajo de la interrupcion del final del codigo

            mov ch, 4ch
            int 21h

            ; aca se los define
            define_clear_screen

======== .model ========

    ; Define el tamaño de los segmentos de datos y de codigo que se va a utilizar

    ;Sintaxis: .model [tamaño]
    ;Variantes comunes:

        ;tiny: El codigo y los datos deben caber en un solo segmento de 64 KB 
        ;small: Es el más usado. El código tiene su propio segmento (64 KB)    
                y los datos el suyo (64 KB)                                    
        ;medium: El codigo puede ser muy grande (varios segmentos), pero los    
                datos deben caber en uno solo                               
        ;large: Tanto el codigo como los datos pueden ocupar varios segmentos  
        

======== .stack ========

    ; Sintaxis: .stack [tamaño en bytes]
    ; Si pones solo .stack, el ensamblador reserva automaticamente 1024 bytes

    ; La pila es una estructura LIFO (Last In, First Out). En 8086, la pila crece hacia abajo en la memoria, 
    ; por lo que al meter datos, el puntero sp disminuye

    PUSH (Empujar): Guarda un valor de 16 bits en la pila y resta 2 a SP
    POP (Sacar): Recupera el último valor guardado, lo pone en un registro y suma 2 a SP

    ; Uso principal: Preservar registros. Si se va a usar ax en un procedimiento pero no queremos perder su valor original, 
    ; se hace un push ax al inicio y un pop ax al final

======== .data ========

    ; Tamaños de variables ------------

    ; db (define byte): Reserva 8 bits y su rango va de 0 a 255 (decimal)       
    ; dw (define word): Reserva 16 bits y su rango va de 0 a 65 535 (decimal)    
    ; dd (define double word): Reserva 32 bit y su rango va de 0 a 4 294 967 295       
    
    ; Ejemplo de definiciones en .data ------------

    var1 db 200d 
    var2 dw 60000d
    var3 dd 90000d
    var4 db ? ; Si no tengo un contenido todavia

    ;Vectores con valores: Se definen separando valores por comas
    precios db 10, 20, 30, 40 ; Crea una lista de 4 bytes contiguos

    ;Vectores vacios
    vector2 db 10 dup(?) ;Crea 10 espacios de 1 byte cada uno sin valor inicial

    ;Cadenas de texto: Se encierran en comillas

    == Etiquetas y saltos ==

    ; En ensamblador se leen las instrucciones de arriba hacia abajo. Para romper ese orden, se usan etiquetas y saltos

    ; Etiquetas ---------------

    ; Sintaxis: Se escriben en minusculas seguidas de dos puntos (ejemplo --> etiqueta1:, fin:, etc)
    ; Proposito: Sirven como puntos de referencia para los saltos; no ocupan espacio en la memoria final, solo guian al compilador

    ; Ejemplo
        Etiqueta1:
            Instruccion 1
            Instruccion 2
            Instruccion 3

        Etiqueta2:
            Instruccion 4
            Instruccion 5

    ; Saltos ------------------

    ; Instrucciones que mueven el flujo del programa hacia una etiqueta especifica

    ; # Comparacion (cmp): Paso previo a un salto condicional. Compara dos valores restandolos internamente para activar banderas

        ; Ejemplo
        cmp al, 13d  ; Compara si al tiene el valor 13d

    ; # Saltos Condicionales: Solo se ejecutan si el resultado de cmp cumple la condición

        ; JE (Jump if Equal): Salta si los valores son iguales    
        ; JNE (Jump Not Equal): Salta si son diferentes             
        ; JG (Jump if Greater): Salta si el primero es mayor        
        ; JGE (J. Greater Equal): Salta si es mayor o igual           
        ; JL (Jump if Less): Salta si el primero es menor        
        ; JLE (J. Less Equal): Salta si es menor o igual           
        ; JA (Jump Above): Superior (para números sin signo)   
        ; JB (Jump Below): Inferior (para números sin signo)   
        
    ; # Salto Incondicional (jmp): Salta a la etiqueta indicada sin evaluar ninguna condición

        ; Ejemplo
        jmp Etiqueta1

    == Vectores  ==

    ; A diferencia de otros lenguajes, no se accede de forma estatica a un vector.
    ; Se usa un registro de indice para "señalar" la posición

    ; Indice si (Source Index): Se usa tradicionalmente para leer un vector de origen
    ; Indice di (Destination Index): Se usa para escribir en un vector de destino

    ; Se inicializan siempre los que se vayan a usar

        xor si, si       ; empezamos en la posicion 0 del origen
        xor di, di       ; empezamos en la posicion 0 del destino
        mov cx, 10d      ; contador para recorrer 10 elementos

    ; No se pueden mover datos directamente de un vector a otro 
        mov vector2, vector1 ; error

    ; Para mover datos de un vector se usan registros intermedios, gereralmente al, bl o dl
        mov bl, vector1[si] ; Paso dato al registro
        mov vector2[di], bl ; Paso el registro al nuevo vector

    == Direccionamiento ==  

        ; Definiciones y cuando se calcula
        ;   OFFSET es Estatico (Fijo): El compilador la calcula ANTES de que el programa se ejecute                       
        ;   LEA es Dinamico (Variable):  El procesador (CPU) la calcula MIENTRAS el programa esta corriendo                  

        ; OFFSET:
        ; Se usa cuando la direccion es fija y no va a cambiar
        ; Uso comun: mov dx, offset vector
        ; Limitacion: Solo acepta nombres de variables. No se puede hacer offset vector[si]

        ; LEA:
        ; Se usa cuando la direccion depende de un registro (como un indice) o algun calculo
        ; Uso comun: lea dx, mi_vector[si]
        ; Que hace: El procesador suma la "Direccion Base" + el "Indice" en tiempo real para saber exactamente a que posicion del vector apuntar

    == Bucles ==

    ; Condicional o manual -----------------

    ; Es el que se construye usando etiquetas y saltos condicionales. Se usa cuando no sabemos cuántas veces se repetirá el proceso 
    ;   (como leer un teclado hasta que presionen "Enter")

    ; Ejemplo: esperar que se ingrese un punto

        esperar_punto:
            mov ah, 01h          ; servicio para leer un caracter
            int 21h              ; el caracter queda en al
            
            cmp al, "."          ; ¿el usuario escribio un punto? 
            je salir_del_bucle   ; si es igual (je), saltamos al final 
            
            jmp esperar_punto    ; si no fue punto, saltamos al inicio para repetir 

        salir_del_bucle:

    ; Contador ---------------

    ; Mas eficiente cuando sabemos exactamente el numero de repeticiones. Utiliza el registro cx como contador

    ; Al ejecutar la instruccion loop, el procesador resta 1 a cx automaticamente. Si cx no es cero, vuelve a la etiqueta
    ;   si es cero, continua con la siguiente línea

        mov cx, 10d          ; queremos repetir 10 veces
        inicio_bucle:

            instruccion1    ; instrucciones a repetir
            instruccion2 

        loop inicio_bucle    ; decrementa cx y salta a inicio_bucle si cx > 0

    ; Indices ---------------

    ; Se usa para recorrer cadenas o arreglos

    ; Ejemplo: recorrer una palabra de 4 bits e imprimir

        .data
            palabra db "hola"    ; nuestra lista de datos
            largo   db 4d        ; sabemos que mide 4

        .code
            xor si, si           ; el indice empieza en la posición 0 

            imprimir_lista:
                mov dl, palabra[si]  ; cargamos en dl el caracter que señala si 
                mov ah, 02h          ; servicio para mostrar caracter
                int 21h              ; imprime la letra actual
                
                inc si               ; movemos el indice a la siguiente posición 
                
                cmp si, 4d           ; ¿ya llegamos a la posición 4 (el final)?
                jne imprimir_lista   ; si no es igual (jne), repetimos el proceso

    == Interrupciones ==

        ; Para usar uno, hay que tener en cuenta estos tres pasos:

        ; 1) ah: Se pone el numero del servicio especifico que se quiere usar
        ; 2) Registros de datos: Se prepara la información necesaria 
            (que imprimir o donde guardar lo leido)
        ; 3) int: Ejecutas la instruccion seguida del numero de interrupcion 
            (ej: int 21h)

        ; Servicios comunes -------------

        ; 01h (Leer Caracter) --> Resultado queda en AL               
        ; 02h (Imprimir Caracter) --> Poner el caracter en DL antes       
        ; 09h (Imprimir Cadena) --> Direccion en DX (debe terminar en $)
        ; 0Ah (Leer Cadena) --> Poner direccion del buffer en DX    
        ; 4Ch (Terminar Prog.) --> Obligatorio para salida limpia      
        

        ; Implementacion en codigo ---------------

        ; UN SOLO CARACTER
        ; Se inicializa el segmento de datos para poder acceder a las variables
            mov dx, @DATA   
            mov ds , dx
            xor dx,dx   

        ; Leer un caracter
            mov ah, 01h      ; Servicio para leer carácter con eco (lo muestra mientras escribes)
            int 21h          ; El carácter queda guardado automáticamente en el registro al

        ; Mostrar un caracter
            mov dl, al       ; Movemos lo que leímos (que está en al) a dl (requisito del servicio 02h)
            mov ah, 02h      ; Servicio para mostrar un carácter en pantalla
            int 21h

        ; Imprimir salto de linea
            mov ah,2h
            mov dl,10d  ; 10d es el salto de linea en decimal en ASCCI
            int 21h

        ; UNA CADENA DE CARACTERES (usando etiquetas e indices)
        ; Obtener una cadena

            xor si, si       ; ponemos el índice en cero
            getchar:

                mov ah, 01h  ; pedimos caracter
                int 21h
                
                cmp al, 13d  ; ¿el usuario presionó enter (ascii 13)? 
                je fingetchar ; si es así, terminamos de leer 
                
                mov string[si], al ; guardamos el caracter en el arreglo 
                inc si       ; movemos el índice a la siguiente posición 
                jmp getchar  ; repetimos para el siguiente caracter

            fingetchar:

            ; al terminar, 'si' contiene el largo de la cadena 

        ; Imprimir la cadena

            xor di, di       ; usamos di para recorrer desde el inicio
            printchar:

                mov ah, 02h  ; servicio para mostrar un caracter
                mov dl, string[di] ; cargamos el caracter actual 
                int 21h
                
                inc di       ; siguiente posición 
                cmp di, si   ; ¿ya imprimimos todos los caracteres leídos? 
                je finprintchar ; si terminamos, saltamos al fin
                
                jmp printchar ; si no, seguimos imprimiendo

            finprintchar:

    == Procedimientos ==

    ;proc: Indica el inicio de un procedimiento
    ;endp: Indica el final del procedimiento
    ;call: Se usa para saltar al procedimiento y ejecutarlo
    ;ret: Es la instruccion más importante; le dice al procesador que regrese al lugar exacto donde lo llamaron

    ; Como usarlo ...

    ; Definición fuera del main --------------
        un_nombre proc

            instruccion1
            instruccion2
            ret              ; vuelve a donde fue llamado

        un_nombre endp

    ; Uso dentro del main --------------
        main proc

            ; ... instrucciones antes...
            call un_nombre ; ejecuta el salto y vuelve aquí
            ; ... instrucciones despues ...

        main endp
